add_executable(Lab3CPP main.cpp)
target_include_directories(Lab3CPP PRIVATE ${CMAKE_SOURCE_DIR}/LibraryCPP)
target_link_libraries(Lab3CPP LibraryCPP)
configure_file(input.txt ${CMAKE_CURRENT_BINARY_DIR}/input.txt COPYONLY)
configure_file(input2.txt ${CMAKE_CURRENT_BINARY_DIR}/input2.txt COPYONLY)
enable_testing()
add_test(NAME Test_Input COMMAND Lab3CPP input.txt)
set_tests_properties(Test_Input PROPERTIES PASS_REGULAR_EXPRESSION "9")
add_test(NAME Test_Input2 COMMAND Lab3CPP input2.txt)
set_tests_properties(Test_Input2 PROPERTIES PASS_REGULAR_EXPRESSION "6")
